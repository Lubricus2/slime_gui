struct with Value and Unit

enum Unit {
    Pixels(f32),      // Standard hardware pixels
    Points(f32),      // DPI-aware (1/72 inch)
    Mm(f32),		  // DPI-aware (mm)
    Em(f32),          // Multiplier of current font size
    Percent(f32),     // % of parent's inner dimension  or percent of the screen or percent of the window.
    Available(f32),   // % of space left AFTER fixed-size siblings  // use Weigth only  
    Weight(f32),      // Flex-style: "Take 1 part of the total available"
}
